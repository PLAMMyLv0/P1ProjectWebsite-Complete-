บทเรียนที่ 3: การทำงานกับ JavaScript

เข้าใจพื้นฐานของ JavaScript และวิธีใช้เพื่อปรับปรุงประสบการณ์ผู้ใช้บนเว็บไซต์.

ทำแบบทดสอบ
หน้าหลัก

เนื้อหา JavaScript

let x = 10;

กำหนดตัวแปร x และตั้งค่าเป็น 10

const PI = 3.14;

กำหนดค่าคงที่ PI เท่ากับ 3.14

function greet(name) { return 'Hello, ' + name; }

นิยามฟังก์ชันที่คืนค่าทักทาย

if (x > 5) { console.log('x is greater than 5'); }

ตรวจสอบค่าของ x และพิมพ์ข้อความหาก x มากกว่า 5

for (let i = 0; i < 5; i++) { console.log(i); }

ใช้ลูป for เพื่อลูปจาก 0 ถึง 4

let array = [1, 2, 3];

สร้างอาร์เรย์ที่มีค่าตัวเลข 1, 2, และ 3

array.push(4);

เพิ่มค่า 4 ไปที่ตอนท้ายของอาร์เรย์

console.log(array);

พิมพ์ค่าอาร์เรย์ไปที่คอนโซล

document.getElementById('myElement').innerText = 'Hello';

เปลี่ยนข้อความขององค์ประกอบที่มี ID 'myElement' เป็น 'Hello'
